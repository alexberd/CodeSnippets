##The following code searches Column for the MyTextToSearch Text
##Reports the previous two and the following two words

SELECT
MyTextColumn, CONCAT(ExtractedTextLeft,"MyTextToSearch", ExtractedTextRight)
FROM
(
	SELECT
	MyTextColumn,
	SUBSTRING_INDEX(SUBSTRING_INDEX(MyTextColumn,"view", 1), " ", -3) AS ExtractedTextLeft,
	SUBSTRING_INDEX(SUBSTRING_INDEX(MyTextColumn,"view", -1), " ", 2) AS ExtractedTextRight
	FROM MyTable
	Where MyTextColumn like "%view%"
)as s;
